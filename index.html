async function fetchLeads() { return new Promise((resolve) => { const iframeId = 'fetch_iframe_' + Date.now(); const iframe = document.createElement('iframe'); iframe.name = iframeId; iframe.id = iframeId; iframe.style.display = 'none'; document.body.appendChild(iframe); const form = document.createElement('form'); form.method = 'GET'; form.action = SCRIPT_URL; form.target = iframeId; form.style.display = 'none'; document.body.appendChild(form); iframe.onload = () => { try { let responseText = ''; try { responseText = iframe.contentDocument?.body?.innerText || ''; } catch(e) { responseText = ''; } let data = { leads: [] }; if (responseText && responseText.trim().startsWith('{')) { data = JSON.parse(responseText); } resolve(data.leads || []); } catch(e) { console.error('Error parsing fetch response', e); resolve([]); } setTimeout(() => { form.remove(); iframe.remove(); }, 100); }; form.submit(); setTimeout(() => { if (form.parentNode) { form.remove(); iframe.remove(); resolve([]); } }, 8000); }); }
